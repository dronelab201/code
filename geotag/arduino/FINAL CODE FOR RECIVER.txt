#include <SPI.h>
#include <LoRa.h>

#define LORA_SS 27
#define LORA_RST 14
#define LORA_DIO0 26

void setup() 
{
  Serial.begin(115200);

  LoRa.setPins(LORA_SS, LORA_RST, LORA_DIO0);

  if (!LoRa.begin(433E6)) 
  {
    Serial.println("LoRa init failed");
    while (1);
  }

  Serial.println("Receiver Ready");
}

void loop() 
{
  int packetSize = LoRa.parsePacket();

  if (packetSize) 
  {
    Serial.print("Received: ");

    while (LoRa.available()) 
    {
      Serial.print((char)LoRa.read());
    }

    Serial.println();
  }
}
